\subsubsection{余数的性质}

\paragraph{余同取余} 若 $x$ 除以 $m$ 余 $a$，$y$ 除以 $n$ 余 $a$，则有：

\[
	x = k\gcd (m,n) + a, \forall k \in Z
\]

\paragraph{和同加和} 若 $x$ 除以 $m$ 余 $a$，$y$ 除以 $n$ 余 $b$，且 $a + m = b + n$，则有：

\[
	x = k\gcd (m,n) + a + m, \forall k \in Z
\]

\paragraph{差同减差} 若 $x$ 除以 $m$ 余 $a$，$y$ 除以 $n$ 余 $b$，且 $a - m = b - n$，则有：

\[
	x = k\gcd (m,n) + a - m, \forall k \in Z
\]
